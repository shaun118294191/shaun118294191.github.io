Hazardous(map)

2020-03-05 2 min read # map合并

给定n个节点，每个节点有一种颜色。对于每个节i, 求i的子树中所有颜色相同的节点两两距离和。

Map合并:

#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef pair<int, int> pii;

#define N 101111

int n;
int color[N], dep[N];

typedef map <int, pair<long long, int>> Map;
Map root[N];
vector <int> G[N];
LL sc[N], delta;
int curdep;

void merge(Map &root, Map &p) {
	if (root.size() < p.size()) swap(root, p);
	for (auto& v : p) {
		int color = v.first;
		auto it = root.find(color);
		if (it != root.end()) {
			LL d1 = it->second.first, c1 = it->second.second;
			LL d2 = v.second.first, c2 = v.second.second;
			delta += d1 * c2 + d2 * c1 - 2ll * curdep * c1 * c2;
			it->second.first += d2;
			it->second.second += int(c2);
		}
		else root.insert(v);
	}
}

void init(Map& root, int color) {
	root[color] = {curdep, 1};
}

void dfs(int x, int fa) {
	curdep = dep[x];
	init(root[x], color[x]);
	for (auto to : G[x]) if (to != fa){
		dep[to] = dep[x] + 1;
		dfs(to, x);
		sc[x] += sc[to];
		delta = 0;
		curdep = dep[x];
		merge(root[x], root[to]);
		sc[x] += delta;
	}
}

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d", color + i);
	for (int i = 1; i < n; i++) {
		int u, v;
		scanf("%d%d", &u, &v);
		G[u].push_back(v);
		G[v].push_back(u);
	}
	dfs(1, -1);
	for (int i = 1; i <= n; i++) printf("%lld ", sc[i]);
}
