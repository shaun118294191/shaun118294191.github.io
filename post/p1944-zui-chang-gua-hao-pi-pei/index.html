P1944 最长括号匹配

2020-03-05 2 min read # 线性dp # 栈

题目链接

求最长括号序列, 若有相同长度的括号序列，则输出最靠前的。

合法括号序列: [], (), [()], [()], ()()[]

  • 用栈来维护下标, 用C[i]表示i处最长括号序列长度。
  • 读入'(' 或 '['就压入栈, 否则判断当前字符是否与栈顶元素匹配。
  • 如果匹配，则有C[i] = i - Stack[top] + 1 + C[Stack[top] - 1]，并弹出栈顶元素。示例 ()
  • 若不匹配，则清空栈。示例[)]，不清空则会出错。
    for (int i = 1; i < s.size(); i++) {
        if (s[i] == '(' || s[i] == '[') Stack[++top] = i;
        else if (top) {
            if ((s[i] == ']' && s[Stack[top]] == '[') ||
                (s[i] == ')' && s[Stack[top]] == '(')) {
                C[i] = i - Stack[top] + 1 + C[Stack[top] - 1];
                maxL = max(maxL, C[i]);
                top--;
            } else while(top) top--;
        }
    }

最后遍历字符串找到第一个C[i] = maxLen的位置，输出即可。

Code:

#include <bits/stdc++.h>
constexpr int MAXN = 1e6 + 6;
constexpr int MAXM = 1e4 + 4;
using namespace std;
template <typename T>

inline void read(T &x) {
    int f = 1;
    x = 0;
    char ch = getchar();
    while (!isdigit(ch)) f = (ch == '-') ? -1 : 1, ch = getchar();
    while (isdigit(ch)) x = x * 10 + ch - 48, ch = getchar();
    x *= f;
}

string s, cur, ans;
int Stack[MAXN], top, C[MAXN], maxL(-1);

int main() {
    std::ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    cin >> s, s = " " + s;
    for (int i = 1; i < s.size(); i++) {
        if (s[i] == '(' || s[i] == '[') Stack[++top] = i;
        else if (top) {
            if ((s[i] == ']' && s[Stack[top]] == '[') ||
                (s[i] == ')' && s[Stack[top]] == '(')) {
                C[i] = i - Stack[top] + 1 + C[Stack[top] - 1];
                maxL = max(maxL, C[i]);
                top--;
            } else while(top) top--;
        }
    }
    for (int i = 1; i < s.size(); i++) {
        if (C[i] == maxL) {
            for (int j = i - maxL + 1; j <= i; j++)cout << s[j];
            return 0;
        }
    }
    return 0;
}
